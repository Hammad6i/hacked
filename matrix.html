const canvas = document.getElementById("matrix"); const ctx = canvas.getContext("2d"); canvas.height = window.innerHeight; canvas.width = window.innerWidth; const chars = "アァイゥエカキクサシスセタチトナニハヒフ0123456789"; const columns = canvas.width / 20; const drops = Array.from({ length: columns }).fill(1); function drawMatrix() { ctx.fillStyle = "rgba(0, 0, 0, 0.05)"; ctx.fillRect(0, 0, canvas.width, canvas.height); ctx.fillStyle = "#0F0"; ctx.font = "16px monospace"; for (let i = 0; i < drops.length; i++) { const text = chars.charAt(Math.floor(Math.random() * chars.length)); ctx.fillText(text, i * 20, drops[i] * 20); if (drops[i] * 20 > canvas.height || Math.random() > 0.95) { drops[i] = 0; } drops[i]++; } } setInterval(drawMatrix, 33);
